% ── Key Findings ─────────────────────────────────────────────
\begin{tcolorbox}[posterbox, title=Key Findings]
  \color{bodytext}
  \begin{itemize}
    \setlength\itemsep{5pt}
    \item \textbf{Modified prompt} reduced prior-knowledge reliance on
      Bernoulli's Principle, improving exact accuracy from 0\% to 67\%
      and RMSLE from 1.18 to 0.17
    \item \textbf{Iterative optimizer} reduced RMSLE from 10.86 to 0.00
      on sound speed in a single optimization cycle, shifting the agent
      from 0 experiments to 18+ per trial
    \item \textbf{Wikipedia scraper} extracted and validated \textbf{21
      new physical laws} across 8 domains, ready for benchmark extension
  \end{itemize}
\end{tcolorbox}
